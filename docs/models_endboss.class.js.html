/**
 * Class representing the Endboss character in the game.
 * @extends MovableObject
 */
class Endboss extends MovableObject {
    height = 500;
    width = 300;
    y = -20;
    speed = 10;
    health = 2;

    IMAGES_WALKING = [
        "img/4_enemie_boss_chicken/1_walk/G1.png",
        "img/4_enemie_boss_chicken/1_walk/G2.png",
        "img/4_enemie_boss_chicken/1_walk/G3.png",
        "img/4_enemie_boss_chicken/1_walk/G4.png",
    ];

    IMAGES_ALERT = [
        "img/4_enemie_boss_chicken/2_alert/G5.png",
        "img/4_enemie_boss_chicken/2_alert/G6.png",
        "img/4_enemie_boss_chicken/2_alert/G7.png",
        "img/4_enemie_boss_chicken/2_alert/G8.png",
        "img/4_enemie_boss_chicken/2_alert/G9.png",
        "img/4_enemie_boss_chicken/2_alert/G10.png",
        "img/4_enemie_boss_chicken/2_alert/G11.png",
        "img/4_enemie_boss_chicken/2_alert/G12.png",
    ];

    IMAGES_ATTACK = [
        "img/4_enemie_boss_chicken/3_attack/G13.png",
        "img/4_enemie_boss_chicken/3_attack/G14.png",
        "img/4_enemie_boss_chicken/3_attack/G15.png",
        "img/4_enemie_boss_chicken/3_attack/G16.png",
        "img/4_enemie_boss_chicken/3_attack/G17.png",
        "img/4_enemie_boss_chicken/3_attack/G18.png",
        "img/4_enemie_boss_chicken/3_attack/G19.png",
        "img/4_enemie_boss_chicken/3_attack/G20.png",
    ];

    IMAGES_HURT = [
        "img/4_enemie_boss_chicken/4_hurt/G21.png",
        "img/4_enemie_boss_chicken/4_hurt/G22.png",
        "img/4_enemie_boss_chicken/4_hurt/G23.png",
    ];

    IMAGES_DEAD = [
        "img/4_enemie_boss_chicken/5_dead/G24.png",
        "img/4_enemie_boss_chicken/5_dead/G25.png",
        "img/4_enemie_boss_chicken/5_dead/G26.png",
    ];

    /**
     * Creates an instance of Endboss.
     */
    constructor() {
        super().loadImage('img/4_enemie_boss_chicken/2_alert/G5.png');
        this.loadImages(this.IMAGES_WALKING);
        this.loadImages(this.IMAGES_ALERT);
        this.loadImages(this.IMAGES_ATTACK);
        this.loadImages(this.IMAGES_HURT);
        this.loadImages(this.IMAGES_DEAD);
        this.x = 3600;
        this.animate();
    }

    /**
     * Animates the endboss character.
     */
    animate() {
        this.animationInterval = setInterval(() => {
            let distance = this.calculatedDistance();

            if (this.isDead()) {
                this.endbossDead();
            } else if (distance <= 200) {
                this.endbossAttack();
            } else if (distance <= 550) {
                this.endbossWalk();
            } else {
                this.endbossAlert();
            }
        }, 200);
    }

    /**
     * Calculates the distance between the endboss and the character.
     * @returns {number} The distance between the endboss and the character.
     */
    calculatedDistance() {
        if (this.world && this.world.character) {
            return Math.abs(this.x - this.world.character.x);
        }
        return Infinity;
    }

    /**
     * Sets the endboss to alert state.
     */
    endbossAlert() {
        this.playAnimation(this.IMAGES_ALERT);
    }

    /**
     * Sets the endboss to walk state and moves it to the left.
     */
    endbossWalk() {
        this.playAnimation(this.IMAGES_WALKING);
        this.moveLeft();
    }

    /**
     * Sets the endboss to attack state, plays the attack sound, and moves it to the left.
     * If the endboss is touching the character, it kills the character and triggers the lost game event.
     */
    endbossAttack() {
        this.playAnimation(this.IMAGES_ATTACK);
        audioManager.play('endboss');
        this.moveLeft();
        if (this.isTouchingCharacter()) {
            this.world.character.die();
            window.lostGame();
        }
    }

    /**
     * Checks if the endboss is touching the character.
     * @returns {boolean} True if the endboss is touching the character, false otherwise.
     */
    isTouchingCharacter() {
        let distance = this.calculatedDistance();
        return distance <= 100;
    }

    /**
     * Sets the endboss to dead state and removes it from the game after a delay.
     */
    endbossDead() {
        if (this.isDead()) {
            return;
        }
        this.playAnimation(this.IMAGES_DEAD);
        setTimeout(() => {
            this.removeEndboss();
            window.winGame();
        }, 1000);
    }

    /**
     * Reduces the health of the endboss by 1. If the health drops to 0, it dies.
     * Otherwise, it plays the hurt animation and sound.
     */
    hit() {
        if (this.isDead()) return;
        this.health -= 1;
        if (this.health <= 0) {
            this.die();
        } else {
            audioManager.play('endbossHurt');
            this.playAnimation(this.IMAGES_HURT);
        }
    }

    /**
     * Sets the endboss to dead state, stops its movement, and triggers the win game event after a delay.
     */
    die() {
        if (this.isDead()) {
            return;
        }
        this.loadImage('img/4_enemie_boss_chicken/5_dead/G24.png');
        setTimeout(() => {
            this.remove();
            this.removeEndboss();
            if (!this.world.winGame) {
                this.world.winGame = true;
                this.world.gameOver = true;
                window.winGame();
            }
        }, 1000);
    }

    /**
     * Removes the endboss from the game.
     */
    removeEndboss() {
        let index = this.world.level.enemies.indexOf(this);
        if (index > -1) {
            this.world.level.enemies.splice(index, 1);
        }
    }
}
