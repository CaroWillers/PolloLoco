/**
 * Class representing a throwable object.
 * @extends MovableObject
 */
class ThrowableObject extends MovableObject {
    IMAGES_BOTTLE_ROTATION = [
        'img/6_salsa_bottle/bottle_rotation/1_bottle_rotation.png', 
        'img/6_salsa_bottle/bottle_rotation/2_bottle_rotation.png',
        'img/6_salsa_bottle/bottle_rotation/3_bottle_rotation.png',
        'img/6_salsa_bottle/bottle_rotation/4_bottle_rotation.png'
    ];

    IMAGES_BOTTLE_SPLASH = [
        'img/6_salsa_bottle/bottle_rotation/bottle_splash/1_bottle_splash.png',
        'img/6_salsa_bottle/bottle_rotation/bottle_splash/2_bottle_splash.png',
        'img/6_salsa_bottle/bottle_rotation/bottle_splash/3_bottle_splash.png',
        'img/6_salsa_bottle/bottle_rotation/bottle_splash/4_bottle_splash.png',
        'img/6_salsa_bottle/bottle_rotation/bottle_splash/5_bottle_splash.png',
        'img/6_salsa_bottle/bottle_rotation/bottle_splash/6_bottle_splash.png'
    ];

    /**
     * Creates a throwable object.
     * @param {number} x - The initial x position of the object.
     * @param {number} y - The initial y position of the object.
     * @param {World} world - The game world instance.
     */
    constructor(x, y, world) {
        super();
        this.loadImage('img/6_salsa_bottle/1_salsa_bottle_on_ground.png');  
        this.loadImages(this.IMAGES_BOTTLE_ROTATION);
        this.loadImages(this.IMAGES_BOTTLE_SPLASH);
        this.x = x;
        this.y = y;
        this.speedY = 15;
        this.world = world;
        this.isRemoved = false;
    }

    /**
     * Initiates the throw action, applying gravity and setting intervals for movement and animation.
     */
    throw() {
        this.applyGravity();
        this.movementInterval = setInterval(() => {
            this.x += 10;
            this.checkGroundCollision();
        }, 25);
        this.animationInterval = setInterval(() => {
            this.playAnimation(this.IMAGES_BOTTLE_ROTATION);
        }, 50);
    }

    /**
     * Checks for ground collision and plays the splash animation if the object hits the ground.
     */
    checkGroundCollision() {
        let splashHeight = 340;
        if (this.y >= splashHeight) {
            clearInterval(this.animationInterval);
            clearInterval(this.movementInterval);
            this.y = splashHeight;
            this.playAnimation(this.IMAGES_BOTTLE_SPLASH);
            audioManager.play('brokenBottle');
            setTimeout(() => {
                this.removeBottle();
            }, 500);
        }
    }

    /**
     * Checks for collisions with enemies and handles the collision event.
     * @param {Array} enemies - The list of enemies to check for collisions.
     */
    checkCollision(enemies) {
        if (this.isRemoved) return;
        enemies.forEach((enemy) => {
            if (this.isColliding(enemy)) {
                clearInterval(this.animationInterval);
                clearInterval(this.movementInterval);
                this.isRemoved = true;
                if (enemy instanceof Endboss) {
                    this.world.handleBottleHitEndboss(this);
                } else {
                    enemy.die();
                    this.playAnimation(this.IMAGES_BOTTLE_SPLASH);
                    audioManager.play('brokenBottle'); 
                    setTimeout(() => {
                        this.removeBottle();
                    }, 500);
                }
            }
        });
    }

    /**
     * Removes the bottle from the game world.
     */
    removeBottle() {
        this.isRemoved = true;
        let index = this.world.throwableObjects.indexOf(this);
        if (index > -1) {
            this.world.throwableObjects.splice(index, 1);
        }
    }
}
