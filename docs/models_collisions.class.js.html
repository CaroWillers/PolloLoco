/**
 * Manages collisions in the game world.
 */
class CollisionManager {
    /**
     * Creates an instance of CollisionManager.
     * @param {World} world - The game world object.
     */
    constructor(world) {
        this.world = world;
    }

    /**
     * Checks all types of collisions in the game world.
     */
    checkCollisions() {
        if (this.world.gameOver) return;
        this.collectCoins();
        this.collectBottles();
        this.checkBottleCollisions();
        this.checkJumpCollisions();
        this.checkWalkCollisions();
        this.checkEndbossCollisions();
    }

    /**
     * Checks for collisions between the character and coins.
     * If a collision is detected, the coin is collected.
     */
    collectCoins() {
        if (this.world.gameOver) return;
        this.world.level.coins.forEach((coin, index) => {
            if (this.world.character.isColliding(coin)) {
                this.world.coinCounter++;
                let percentage = (this.world.coinCounter / 5) * 100;
                this.world.statusBarCoin.setPercentage(percentage);
                this.world.level.coins.splice(index, 1);
                audioManager.play('coin', 2000);
            }
        });
    }

    /**
     * Checks for collisions between the character and bottles.
     * If a collision is detected, the bottle is collected.
     */
    collectBottles() {
        if (this.world.gameOver) return;
        this.world.level.bottle.forEach((bottle, index) => {
            if (this.world.character.isColliding(bottle)) {
                this.world.bottleCounter++;
                let percentage = (this.world.bottleCounter / 5) * 100;
                this.world.statusBarBottle.setPercentage(percentage);
                this.world.character.collectedBottles++;
                audioManager.play('bottle', 2000);
                this.world.level.bottle.splice(index, 1);
            }
        });
    }

    /**
     * Checks for collisions between throwable objects (bottles) and enemies or the ground.
     */
    checkBottleCollisions() {
        if (this.world.gameOver) return;
        this.world.throwableObjects.forEach((bottle) => {
            bottle.checkCollision(this.world.level.enemies);
            bottle.checkGroundCollision();
        });
    }

    /**
     * Checks for jump collisions between the character and enemies.
     * If a collision is detected, the enemy is defeated.
     */
    checkJumpCollisions() {
        if (this.world.gameOver) return;
        this.world.level.enemies.forEach((enemy) => {
            if (this.world.character.isAboveGround() && this.world.character.isColliding(enemy) && this.world.character.y + this.world.character.height < enemy.y + enemy.height / 2) {
                enemy.die();
                audioManager.play('killedEnemy', 2000);
                this.world.character.playAnimation(this.world.character.IMAGES_IDLE);
            }
        });
    }

    /**
     * Checks for walk collisions between the character and enemies.
     * If a collision is detected, the character takes damage.
     */
    checkWalkCollisions() {
        if (this.world.gameOver || this.world.character.jumpCooldown) return;
        this.world.level.enemies.forEach((enemy) => {
            if (this.world.character.isColliding(enemy)) {
                this.world.character.hit();
                this.world.statusBarHealth.setPercentage(this.world.character.health);
                if (this.world.character.isDead()) {
                    this.world.character.die();
                    if (!this.world.lostGame) {
                        this.world.lostGame = true;
                        this.world.gameOver = true;
                        window.lostGame();
                    }
                }
            }
        });
    }

    /**
     * Checks for collisions between the character and the endboss.
     * If a collision is detected, the character takes damage.
     */
    checkEndbossCollisions() {
        if (this.world.gameOver) return;
        if (this.world.level.endboss && this.world.character.isColliding(this.world.level.endboss)) {
            if (!this.world.character.isJumping) {
                this.world.character.hit();
                this.world.statusBarHealth.setPercentage(this.world.character.health);
                if (this.world.character.isDead()) {
                    this.world.character.die();
                    if (!this.world.lostGame) {
                        this.world.lostGame = true;
                        this.world.gameOver = true;
                        window.lostGame();
                    }
                }
            }
        }
    }

    /**
     * Handles the collision between a thrown bottle and the endboss.
     * @param {ThrowableObject} bottle - The bottle that collided with the endboss.
     */
    handleBottleHitEndboss(bottle) {
        if (this.gameOver) return;
        if (this.level.endboss) {
            this.level.endboss.hit();
            bottle.removeBottle();

            if (this.level.endboss.health > 0) {
                this.level.endboss.playAnimation(this.level.endboss.IMAGES_HURT);
                audioManager.play('endbossHurt', 2000);
            }

            if (this.level.endboss.health <= 0) {
                this.level.endboss.die();
                if (!this.winGame) {
                    this.winGame = true;
                    this.gameOver = true;
                    window.winGame();
                }
            }
        }
    }
}
