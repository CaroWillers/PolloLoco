/**
 * Class representing a coin in the game.
 * @extends MovableObject
 */
class Coins extends MovableObject {
    IMAGES_COIN = [
        'img/8_coin/coin_1.png',
        'img/8_coin/coin_2.png'
    ];

    offset = {
        top: 10,
        left: 10,
        right: 10,
        bottom: 10
    };

    /**
     * Create a coin.
     */
    constructor() {
        super().loadImage("img/8_coin/coin_1.png");
        this.loadImages(this.IMAGES_COIN);
        this.width = 100;
        this.height = 100;
        this.x = 700 + Math.random() * 719 * 4 - 700 + this.offset.left - this.offset.right;
        this.y = 350 - Math.random() * 150 + this.offset.top - this.offset.bottom;
        this.animateCoins();
    }

    /**
     * Animate the coin to rotate between images.
     */
    animateCoins() {
        setInterval(() => {
            this.playAnimation(this.IMAGES_COIN);
        }, 250);
    }
}
