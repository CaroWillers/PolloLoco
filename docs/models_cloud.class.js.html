/**
 * Class representing a Cloud in the game.
 * @extends MovableObject
 */
class Cloud extends MovableObject {
    y = 50;
    height = 250;
    width = 719;

    /**
     * Creates an instance of Cloud.
     */
    constructor() {
        super().loadImage('img/5_background/layers/4_clouds/1.png');
        this.x = Math.random() * this.level_end_x;
        this.animate();
    }

    /**
     * Animates the cloud by moving it to the left.
     * If the cloud moves off the screen, it repositions to the right side of the screen.
     */
    animate() {
        setInterval(() => {
            this.moveLeft();
            if (this.x + this.width < 0) {
                this.x = this.level_end_x;
            }
        }, 1000 / 60);
    }
}
