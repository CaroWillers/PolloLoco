/**
 * Class representing a movable object in the game.
 * @extends DrawableObject
 */
class MovableObject extends DrawableObject {
    /**
     * @property {number} speed - The movement speed of the object.
     * @property {boolean} otherDirection - Indicates if the object is moving in the opposite direction.
     * @property {number} speedY - The vertical speed of the object.
     * @property {number} acceleration - The acceleration of the object.
     * @property {number} health - The health of the object.
     * @property {number} lastHit - The timestamp of the last hit.
     * @property {Object} world - The game world the object belongs to.
     * @property {number} movingInterval - The interval ID for the movement interval.
     * @property {number} animationInterval - The interval ID for the animation interval.
     */
    speed = 0.15;
    otherDirection = false;
    speedY = 0;
    acceleration = 2.5;
    health = 100;  
    lastHit = 0; 
    world;
    movingInterval;
    animationInterval;

    /**
     * Applies gravity to the object.
     */
    applyGravity() {
        setInterval(() => {
            if(this.isAboveGround() || this.speedY > 0 ) {   
                this.y -= this.speedY;
                this.speedY -= this.acceleration; 
            }
        }, 1000 / 25);
    }

    /**
     * Checks if the object is above ground.
     * @returns {boolean} - True if the object is above ground, false otherwise.
     */
    isAboveGround() {
        if (this instanceof Bottle) {  
            return true;
        } else {
            return this.y < 340;
        }
    }

    /**
     * Checks if the object is colliding with another object.
     * @param {Object} obj - The object to check collision with.
     * @returns {boolean} - True if the objects are colliding, false otherwise.
     */
    isColliding(obj) {
        return (this.x + this.offset.left < obj.x + obj.width - obj.offset.right &&
            this.x + this.width - this.offset.right > obj.x + obj.offset.left &&
            this.y + this.offset.top < obj.y + obj.height - obj.offset.bottom &&
            this.height + this.y - this.offset.bottom > obj.y + obj.offset.top);
    }

    /**
     * Handles the object being hit.
     */
    hit() { 
        this.energy -= 5;
        if(this.energy <= 0) {
            this.energy = 0; 
            audioManager.play('hit', 2000);
        } else {
            this.lastHit = new Date().getTime(); 
        }
    }

    /**
     * Checks if the object is hurt.
     * @returns {boolean} - True if the object is hurt, false otherwise.
     */
    isHurt() {  
        let timepassed = new Date().getTime() - this.lastHit;
        timepassed = timepassed / 1000;
        if (timepassed < 1) {
            audioManager.play('hurt', 2000);
            return true;
        }
        return false; 
    }

    /**
     * Checks if the object is dead.
     * @returns {boolean} - True if the object is dead, false otherwise.
     */
    isDead() {
        if (this.health == 0) {
            return true;
        }
        return false;
    }
    
    /**
     * Plays the specified animation.
     * @param {Array<string>} images - The images to animate.
     */
    playAnimation(images) { 
        let i = this.currentImage % images.length; // let i = 7 % 6 => 1, Rest 1
        let path = images[i];
        this.img = this.imageCache[path];
        this.currentImage++;
    }

    /**
     * Moves the object to the right.
     */
    moveRight() {
        this.x += this.speed;
    }

    /**
     * Moves the object to the left.
     */
    moveLeft() {
        this.x -= this.speed;
    }

    /**
     * Makes the object jump.
     */
    jump() {
        this.speedY = 30;
    }

    /**
     * Checks if the object is falling.
     * @returns {boolean} - True if the object is falling, false otherwise.
     */
    isFalling() {
        return this.speedY < 0; 
    }

    /**
     * Handles the object dying.
     */
    die() {
        this.loadImage(this.IMAGES_DEAD[0]);
        this.speed = 0;
        clearInterval(this.movingInterval);
        clearInterval(this.animationInterval);
        setTimeout(() => {
            this.remove();
        }, 1000);
    }

    /**
     * Removes the object from the game world.
     */
    remove() {
        let index = this.world.level.enemies.indexOf(this);
        if (index > -1) {
            this.world.level.enemies.splice(index, 1);
        }
    }
}
